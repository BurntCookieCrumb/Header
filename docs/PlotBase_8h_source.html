235 void Plot::SetProperties(TObject* obj, Int_t index){
236 
239  if (obj->InheritsFrom("TPave")){ //TLegend
240  ((TLegend*)obj)->SetTextFont(font);
241  ((TLegend*)obj)->SetTextSize(label);
242  ((TLegend*)obj)->SetBorderSize(0);
243  return;
244  }
245 
246  if (!styles) return; // no arrays were set, properties were set in advance by hand
247 
248  Double_t size, lwidth; Int_t lstyle, color, marker;
249 
250  size = (index < sizes.size()) ? sizes[index] : 2.;
251  lstyle = (index < lstyles.size()) ? lstyles[index] : 1;
252  lwidth = (index < lwidths.size()) ? lwidths[index] : 2.;
253  color = (index < colors.size()) ? colors[index] : kBlack;
254  marker = (index < markers.size()) ? markers[index] : kFullCircle;
255 
256  if (obj->InheritsFrom("TH1")) {
257  SetPlottjectProperties((TH1*)obj, color, marker, size, lstyle, lwidth);
258  ((TH1*)obj)->SetStats(kFALSE);
259  }
260  else if (obj->InheritsFrom("TF1")){
261  SetPlottjectProperties((TF1*)obj, color, marker, size, lstyle, lwidth);
262  }
263  else if (obj->InheritsFrom("TGraph")){
264  SetPlottjectProperties((TGraph*)obj, color, marker, size, lstyle, lwidth);
265  }
266  else if (obj->InheritsFrom("TMultiGraph")){
267  TIter iMultiGraph(((TMultiGraph*)obj)->GetListOfGraphs());
268  while (TObject* graph = iMultiGraph()){
269  if (!graph) continue;
270  if (index >= markers.size()) break;
271  SetPlottjectProperties((TGraph*)graph, color, marker, size, lstyle, lwidth);
272  index++;
273  }
274  }
275  else if (obj->InheritsFrom("TLine")){
276  SetLineProperties((TLine*)obj, color, lstyle, lwidth);
277  }
278  else{
279  std::cout << "\033[1;34mMissing Class \033[0m" << obj->ClassName() << std::endl;
280  }
