262 void Plot::SetProperties(TObject* obj, Int_t index){
263 
266  Double_t size, lwidth; Int_t lstyle;
267  size = !(sizes.empty()) ? sizes[index] : 2.;
268  lstyle = !(lstyles.empty()) ? lstyles[index] : 1;
269  lwidth = !(lwidths.empty()) ? lwidths[index] : 2.;
270 
271  if (obj->InheritsFrom("TPave")){ //TLegend
272  ((TLegend*)obj)->SetTextFont(font);
273  ((TLegend*)obj)->SetTextSize(label);
274  ((TLegend*)obj)->SetBorderSize(0);
275  }
276 
277  else if (!styles) return;
278  else if (index >= markers.size()) return;
279 
280  else if (obj->InheritsFrom("TH1")) {
281  SetPlottjectProperties((TH1*)obj, colors[index], markers[index], size, lstyle, lwidth);
282  ((TH1*)obj)->SetStats(kFALSE);
283  }
284  else if (obj->InheritsFrom("TF1")){
285  SetPlottjectProperties((TF1*)obj, colors[index], markers[index], size, lstyle, lwidth);
286  }
287  else if (obj->InheritsFrom("TGraph")){
288  SetPlottjectProperties((TGraph*)obj, colors[index], markers[index], size, lstyle, lwidth);
289  }
290  else if (obj->InheritsFrom("TMultiGraph")){
291  TIter iMultiGraph(((TMultiGraph*)obj)->GetListOfGraphs());
292  while (TObject* graph = iMultiGraph()){
293  if (!graph) continue;
294  if (index >= markers.size()) break;
295  SetPlottjectProperties((TGraph*)graph, colors[index], markers[index], size, lstyle, lwidth);
296  index++;
297  }
298  }
299  else if (obj->InheritsFrom("TLine")){
300  SetLineProperties((TLine*)obj, colors[index], lstyle, lwidth);
301  }
302  else{
303  std::cout << "\033[1;34mMissing Class \033[0m" << obj->ClassName() << std::endl;
304  }
305 
306 }
